Я перешел на Linux, поэтому привожу решение для Linux KDE Neon. Эта система установленна у меня на компьютере.

sudo apt install git
git --versioni
git config --global user.name "Sergey Yarkov"
git config --global user.email "channel.my.honey@gmail.com"
git config --list --show-origin

git init
touch .gitignore
nano .gitignore
*.sass
/bin
/admin
/config

git status
git add .
git status
git commit -m "Создал первый Коммит"                
git branch -m main

git branch dev
git checkout dev 
mkdir blog 
cd blog
touch index.html index.js

git checkout main
git remote add origin git@github.com:Oxana-S/itproger.git
git remote -v
git push origin main
